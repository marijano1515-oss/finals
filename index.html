    const todoInput=document.getElementById('todoInput'); const addBtn=document.getElementById('addBtn'); const todoList=document.getElementById('todoList'); const adviceText=document.getElementById('adviceText'); const newAdviceBtn=document.getElementById('newAdviceBtn'); async function getAdvice(){ try{ const resp=await fetch("https://api.adviceslip.com/advice"); if(!resp.ok){ throw new Error("არასწორი მონაცემები"); } const data = await resp.json(); adviceText.textContent=data.slip.advice; }catch(error){ console.log(error); } } window.addEventListener('DOMContentLoaded',getAdvice); newAdviceBtn.addEventListener('click',getAdvice); function addTask(){ const taskText=todoInput.value.trim(); if(taskText===""){ alert("ტექსტი ცარიელია"); return; } const newList=document.createElement('li'); const deleteBtn=document.createElement('span'); deleteBtn.textContent=' ->X'; newList.classList.add('todo-item'); newList.textContent=taskText; newList.appendChild(deleteBtn); newList.addEventListener('click',function(){ newList.classList.toggle('completed'); }) deleteBtn.addEventListener('click',function(ev){ ev.stopPropagation(); const c=confirm("ნამდვილად შლით?"); if(c){ newList.remove(); } }); todoInput.value=""; todoList.appendChild(newList); } addBtn.addEventListener('click',addTask);
